<script>{`
  const el = document.getElementById('demo-tutorial');
  if (el) {
    el.tutorials = [
      {
        title: 'Create your first resource',
        description: 'Learn how to create and configure a resource.',
        completed: false,
      },
      {
        title: 'Set up monitoring',
        description: 'Configure alarms and dashboards.',
        completed: true,
      },
    ];
  }
`}</script>

> **Note:** Tutorials are configured via JavaScript properties. See usage below.

## Usage

```html
<cs-tutorial-panel></cs-tutorial-panel>

<script>
  const panel = document.querySelector('cs-tutorial-panel');
  panel.tutorials = [
    {
      title: 'Create your first resource',
      description: 'Learn how to create and configure a resource.',
      completed: false,
    },
    {
      title: 'Set up monitoring',
      description: 'Configure alarms and dashboards.',
      completed: true,
    },
  ];
</script>
```
