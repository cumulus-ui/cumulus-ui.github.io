  if (el) {
    el.identity = { href: '#', title: 'My Application' };
    el.utilities = [
      { type: 'button', iconName: 'notification', ariaLabel: 'Notifications' },
      { type: 'button', iconName: 'settings', ariaLabel: 'Settings' },
      {
        type: 'menu-dropdown',
        text: 'User',
        iconName: 'user-profile',

        ],
      },
    ];
  }
`}</script>

> **Note:** The `identity` and `utilities` properties are set via JavaScript. See usage below.

## Usage

```html
<cs-top-navigation></cs-top-navigation>

<script>
  const nav = document.querySelector('cs-top-navigation');
  nav.identity = {
    href: '/',
    title: 'My Application',
  };
  nav.utilities = [
    { type: 'button', iconName: 'notification', ariaLabel: 'Notifications' },
    { type: 'button', iconName: 'settings', ariaLabel: 'Settings' },
    {
      type: 'menu-dropdown',
      text: 'User',
      iconName: 'user-profile',
      items: [
        { id: 'profile', text: 'Profile' },
        { id: 'signout', text: 'Sign out' },
      ],
    },
  ];
</script>
```
