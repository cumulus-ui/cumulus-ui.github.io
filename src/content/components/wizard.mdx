<script>{`
  const el = document.getElementById('demo-wizard');
  if (el) {
    const step1 = document.createElement('cs-container');
    step1.setAttribute('header', 'Database engine');
    step1.innerHTML = '<cs-tiles id="wiz-tiles"></cs-tiles>';

    const step2 = document.createElement('cs-container');
    step2.setAttribute('header', 'Instance settings');
    step2.innerHTML = \`
      <cs-space-between size="l" direction="vertical">
        <cs-form-field label="Instance class">
          <cs-select id="wiz-instance" placeholder="Choose instance class"></cs-select>
        </cs-form-field>
        <cs-form-field label="Storage (GiB)">
          <cs-input type="number" value="20"></cs-input>
        </cs-form-field>
      </cs-space-between>
    \`;

    const step3 = document.createElement('cs-container');
    step3.setAttribute('header', 'Review');
    step3.innerHTML = \`
      <cs-space-between size="l" direction="vertical">
        <cs-key-value-pairs id="wiz-review"></cs-key-value-pairs>
      </cs-space-between>
    \`;

    el.steps = [
      { title: 'Choose engine', description: 'Select a database engine for your instance.', content: step1 },
      { title: 'Configure settings', description: 'Set up instance size and storage.', content: step2 },
      { title: 'Review and create', description: 'Verify your configuration before launching.', content: step3 },
    ];
  }
`}</script>
